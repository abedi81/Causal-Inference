#  Synthetic Control (Optimization)
# setting weights (w) that minimize squared error in pre-treatment period
T0 = len(pre_treatment_years)

w = cp.Variable(Y0.shape[1])  # weights for donor states
objective = cp.Minimize(cp.sum_squares(Y1[:T0] - Y0[:T0, :] @ w))
constraints = [cp.sum(w) == 1, w >= 0]  # weights sum to 1, non-negative
problem = cp.Problem(objective, constraints)
problem.solve()

weights = w.value
weights.round(3)
